/**
 * Takes all selected tokens and adds them to the combat tracker. Then rolls initative for all NPC tokens.
 * Install "Hide GM Rolls" Module so players don't see NPC initiative rolls in chat for GM rolls. Less clutter is always good.
*/
async function main() {
  await canvas.tokens.toggleCombat();
// Uncomment this if you want these rolls to be public
// game.combat.rollNPC({ messageOptions: { rollMode: 'publicroll' }})

// Leave this active if you want GM private rolls.
  game.combat.rollNPC({ messageOptions: { rollMode: 'gmroll' }})
}
main();
